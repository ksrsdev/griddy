SDL_Texture* CreateTextureFromText(SDL_Renderer *renderer, TTF_Text *text)
{
    s32 count = 0;
    TTF_SubString **subs = TTF_GetTextSubStringsForRange(text, 0, -1, &count);
    if (!subs || count == 0) return NULL;

    // Use extreme defaults so the first comparison always succeeds
    f32 inkMinX = 1e6f, inkMinY = 1e6f;
    f32 inkMaxX = -1e6f, inkMaxY = -1e6f;

    TTF_Font *font = TTF_GetTextFont(text);
    int fontAscent = TTF_GetFontAscent(font);

    for (s32 i = 0; i < count; i++) {
        const TTF_SubString *sub = subs[i];
        
        // Skip whitespace (they have no 'ink')
        if (sub->flags & TTF_SUBSTRING_WHITESPACE) continue;

        int gMinX, gMaxX, gMinY, gMaxY, gAdvance;
        if (TTF_GetGlyphMetrics(font, sub->cluster, &gMinX, &gMaxX, &gMinY, &gMaxY, &gAdvance)) {
            
            // sub->rect.x/y is the TOP-LEFT of the character's logical box.
            // gMinX/gMaxX are relative to the glyph origin (usually 0).
            // gMinY/gMaxY are relative to the BASELINE.
            
            f32 charMinX = (f32)sub->rect.x + gMinX;
            f32 charMaxX = (f32)sub->rect.x + gMaxX;

            // This is the "magic" vertical math:
            // 1. Move to the logical top (sub->rect.y)
            // 2. Add the font's ascent to get to the baseline
            // 3. Subtract gMaxY (top of ink) and gMinY (bottom of ink)
            // Note: In SDL3_ttf, gMaxY is the distance ABOVE the baseline (positive)
            f32 charMinY = (f32)sub->rect.y + (fontAscent - gMaxY);
            f32 charMaxY = (f32)sub->rect.y + (fontAscent - gMinY);

            inkMinX = SDL_min(inkMinX, charMinX);
            inkMaxX = SDL_max(inkMaxX, charMaxX);
            inkMinY = SDL_min(inkMinY, charMinY);
            inkMaxY = SDL_max(inkMaxY, charMaxY);
        }
    }
    SDL_free(subs);

    // If we only had spaces, or metrics failed
    if (inkMinX > inkMaxX) return NULL;

    int finalW = (int)(inkMaxX - inkMinX);
    int finalH = (int)(inkMaxY - inkMinY);

    SDL_Texture *texture = SDL_CreateTexture(renderer, SDL_PIXELFORMAT_RGBA8888, 
                                            SDL_TEXTUREACCESS_TARGET, finalW, finalH);
    
    // ... Clear Texture Code ...

    // Draw the text shifted so the inkMinX/Y is at (0,0)
    TTF_DrawRendererText(text, -inkMinX, -inkMinY);

    return texture;
}

// 1. Render the text to a temporary surface first
SDL_Surface *surf = TTF_RenderText_Blended(font, text->text, 0, (SDL_Color){255,255,255,255});
if (!surf) return NULL;

int minX = surf->w, minY = surf->h, maxX = 0, maxY = 0;
bool foundPixels = false;

// 2. Lock the surface to read raw pixels
SDL_LockSurface(surf);
u32 *pixels = (u32 *)surf->pixels;

for (int y = 0; y < surf->h; y++) {
    for (int x = 0; x < surf->w; x++) {
        // Check the Alpha channel of the pixel (assuming 32-bit RGBA/ARGB)
        // We look for any pixel that isn't completely transparent
        u32 pixel = pixels[y * (surf->pitch / 4) + x];
        u8 a, r, g, b;
        SDL_GetRGBA(pixel, SDL_GetPixelFormatDetails(surf->format), NULL, &r, &g, &b, &a);

        if (a > 0) { // Found ink!
            if (x < minX) minX = x;
            if (x > maxX) maxX = x;
            if (y < minY) minY = y;
            if (y > maxY) maxY = y;
            foundPixels = true;
        }
    }
}
SDL_UnlockSurface(surf);

if (!foundPixels) {
    SDL_DestroySurface(surf);
    return NULL;
