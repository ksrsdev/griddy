### EXAMPLE SHADER ###

// This is what the GPU runs for every pixel of your text
struct PixelInput {
    float4 position : SV_POSITION;
    float2 texCoord : TEXCOORD0;
    float4 color    : COLOR0;
};

Texture2D s_Texture : register(t0);
SamplerState s_Sampler : register(s0);

float4 main(PixelInput input) : SV_Target {
    // 1. Grab the distance value from the texture
    // SDF fonts usually store the distance in the 'alpha' or 'red' channel
    float distance = s_Texture.Sample(s_Sampler, input.texCoord).r;

    // 2. Define how sharp the edge is. 
    // Higher values = softer/blurry. Lower = razor sharp.
    // For your 'explosion', a tiny bit of softness looks better.
    float smoothing = 0.02; 
    
    // 3. The smoothstep math: 
    // If distance < 0.5 - smoothing, alpha = 0
    // If distance > 0.5 + smoothing, alpha = 1
    float alpha = smoothstep(0.5 - smoothing, 0.5 + smoothing, distance);

    // 4. Multiply by the text color (passed from your C code)
    return float4(input.color.rgb, input.color.a * alpha);
}

// 1. Load the HLSL source (can be a hardcoded string or loaded file)
const char* hlsl_source = "... your SDF fragment shader code ...";

// 2. Initialize ShaderCross
SDL_ShaderCross_Init();

// 3. Compile the shader for the CURRENT renderer's backend
// ShaderCross handles the "Vulkan vs DX12" check automatically here.
SDL_GPUShader* fragShader = SDL_ShaderCross_CompileGraphicsShaderFromHLSL(
    SDL_GetPointerProperty(SDL_GetRendererProperties(renderer), SDL_PROP_RENDERER_GPU_DEVICE_POINTER, NULL),
    &(SDL_ShaderCross_GraphicsShaderInfo){
        .source = hlsl_source,
        .entrypoint = "main",
        .stage = SDL_GPU_SHADERSTAGE_FRAGMENT
    }
);

// 4. Create the Render State
SDL_GPURenderState* sdfState = SDL_CreateGPURenderState(renderer, &(SDL_GPURenderStateCreateInfo){
    .fragment_shader = fragShader,
    // Note: You usually don't need a custom vertex shader for simple texture scaling
});

// Standard SDF Fragment Shader
float4 main(float2 uv : TEXCOORD0) : SV_Target {
    float dist = tex.Sample(samp, uv).a;
    // 0.5 is the edge. fwidth() provides hardware-accelerated anti-aliasing
    float width = fwidth(dist); 
    float alpha = smoothstep(0.5 - width, 0.5 + width, dist);
    return float4(1.0, 1.0, 1.0, alpha); // White text
}
